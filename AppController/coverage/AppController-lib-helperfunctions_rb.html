451   # Queries Amazon EC2's Spot Instance pricing history to see how much other
 
452   # users have paid for the given instance type (assumed to be a Linux box),
 
453   # so that we can place a bid that is similar to the average price. How
 
454   # similar to the average price to pay is a bit of an open problem - for now,
 
455   # we pay 20% more so that in case the market price goes up a little bit, we
 
456   # still get to keep our instances.
 
457   def self.get_optimal_spot_price(instance_type)
 
458     command = "ec2-describe-spot-price-history -t #{instance_type} | " +
 
459       "grep 'Linux/UNIX' | awk '{print $2}'".split("\n")
 
460     prices = `#{command}`
 
461 
462     average = prices.reduce(0.0) { |sum, price|
 
463       sum += Float(price)
 
464     }
 
465     
466     average /= prices.length
 
467     plus_twenty = average * 1.20
 
468     
469     Kernel.puts("The average spot instance price for a #{instance_type} " +
 
470       "machine is $#{average}, and 20% more is $#{plus_twenty}")
 
471     return plus_twenty
 
472   end

666       instances_created << "#{public_ips[index]}:#{private_ips[index]}:#{jobs[index]}:#{instance_ids[index]}:#{cloud}"
 
667     }

669     end_time = Time.now
 
670     total_time = end_time - start_time
 
671 
672     if spot
 
673       Kernel.puts("TIMING: It took #{total_time} seconds to spawn " +
 
674         "#{num_of_vms_to_spawn} spot instances")

677         "#{num_of_vms_to_spawn} regular instances")
 
678     end

871   def self.parse_static_data app_name
 
872     untar_dir = get_untar_dir(app_name)
 
873 
874     begin
 
875       tree = YAML.load_file(File.join(untar_dir,"app.yaml"))
 
876     rescue Errno::ENOENT => e
 
877       Kernel.puts("Failed to load YAML file to parse static data")

887 
888     skip_files_regex = DEFAULT_SKIP_FILES_REGEX
 
889     if tree["skip_files"]

891       input_regex = tree["skip_files"]

895       input_regex.gsub!(/ /,"")

897     end
 
898 
899     handlers.map! do |handler|
 
900       next if !handler.key?("static_dir") && !handler.key?("static_files")

912         FileUtils.mkdir_p cache_static_dir_path

914         filenames = Dir.glob(File.join(untar_dir, handler["static_dir"],"*"))
 
915 
916         # Remove all files which match the skip file regex so they do not get copied
 
917         filenames.delete_if { |f| File.expand_path(f).match(skip_files_regex) }
 
918 
919         FileUtils.cp_r filenames, cache_static_dir_path
 
920 
921         handler["expiration"] = expires_duration(handler["expiration"]) || default_expiration
 
922       elsif handler["static_files"]
 
923         # This is for bug https://bugs.launchpad.net/appscale/+bug/800539
 
924         # this is a temp fix

930         handler["static_files"] = handler["static_files"].gsub(/\\/,"$")
 
931 
932         upload_regex = Regexp.new(handler["upload"])

960   def self.expires_duration input_string
 
961     return nil if input_string.nil? || input_string.empty?
 
962     # Start with nil so we can distinguish between it not being set and 0
 
963     duration = nil
 
964     input_string.split.each do |token|
 
965       match = token.match(DELTA_REGEX)
 
966       next if not match
 
967       amount, units = match.captures
 
968       next if amount.empty? || units.empty?
 
969       duration = (duration || 0) + TIME_IN_SECONDS[units.downcase]*amount.to_i
 
970     end
 
971     duration
 
972   end
