78   # A helper method that makes SOAP calls for us. This method is mainly here to
 
79   # reduce code duplication: all SOAP calls expect a certain timeout and can
 
80   # tolerate certain exceptions, so we consolidate this code into this method.
 
81   # Here, the caller specifies the timeout for the SOAP call (or NO_TIMEOUT
 
82   # if an infinite timeout is required) as well as whether the call should
 
83   # be retried in the face of exceptions. Exceptions can occur if the machine
 
84   # is not yet running or is too busy to handle the request, so these exceptions
 
85   # are automatically retried regardless of the retry value. Typically
 
86   # callers set this to false to catch 'Connection Refused' exceptions or
 
87   # the like. Finally, the caller must provide a block of
 
88   # code that indicates the SOAP call to make: this is really all that differs
 
89   # between the calling methods. The result of the block is returned to the
 
90   # caller. 
91   def make_call(time, retry_on_except, callr, ok_to_fail=false)
 
92     refused_count = 0
 
93     max = 5
 
94 
95     begin
 
96       Timeout::timeout(time) {
 
97         yield if block_given?
 
98       }
 
99     rescue Errno::ECONNREFUSED, Errno::EHOSTUNREACH
 
100       if refused_count > max
 
101         return false if ok_to_fail
 
102         raise Exception.new("Connection was refused. Is the AppController running?")
 
103       else
 
104         refused_count += 1
 
105         Kernel.sleep(1)
 
106         retry
 
107       end
 
108     rescue Timeout::Error
 
109       Kernel.puts("Saw a Timeout Error")
 
110       if ok_to_fail
 
111         return false
 
112       else
 
113         Kernel.sleep(1)
 
114         retry
 
115       end
 
116     rescue OpenSSL::SSL::SSLError, NotImplementedError, Errno::EPIPE, Errno::ECONNRESET
 
117       Kernel.puts("Saw a transient error")
 
118       Kernel.sleep(1)
 
119       retry
 
120     rescue Exception => except
 
121       Kernel.puts("Saw an Exception of class #{except.class}")
 
122       if retry_on_except
 
123         retry
 
124       else
 
125         trace = except.backtrace.join("\n")
 
126         raise Exception.new("[#{callr}] We saw an unexpected error of the " +
 
127           "type #{except.class} with the following message:\n#{except}. " +
 
128           "Trace is #{trace}")
 
129       end
 
130     end
 
131   end
