228   # AppController data in ZooKeeper. This is useful for methods that aren't
 
229   # sure if they already have the ZooKeeper lock or not, but definitely need

232   def self.lock_and_run(&block)

234 
235     # Create the ZK lock path if it doesn't exist.
 
236     NeptuneManager.log("Checking to see if AppController path exists")
 
237     if !self.exists?(APPCONTROLLER_PATH)

239       self.set(APPCONTROLLER_PATH, DUMMY_DATA, NOT_EPHEMERAL)
 
240     end
 
241 
242     # Try to get the lock, and if we can't get it, see if we already have
 
243     # it. If we do, move on (but don't release it later since this block

256         }
 
257         owner = JSON.load(info[:data])
 
258         if @@client_ip == owner
 
259           NeptuneManager.log("Tried to get the lock, but we already have it. " +
 
260             "Not grabbing/releasing lock again.")

274     end

277       "block")
 
278     begin
 
279       yield  # invoke the user's block, and catch any uncaught exceptions
 
280     #rescue Exception => except
 
281     #  NeptuneManager.log("Ran caller's block but saw an Exception of class " +
 
282     #    "#{except.class}")
 
283     #  raise except

285       if got_lock
 
286         NeptuneManager.log("Grabbed lock, so now releasing it.")
 
287         self.release_appcontroller_lock()
 
288         NeptuneManager.log("Released lock successfully")
 
289       else
 
290         NeptuneManager.log("Didn't grab lock, so not releasing it.")
 
291       end
 
292     end
