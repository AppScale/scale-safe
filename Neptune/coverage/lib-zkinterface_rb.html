186   # Gets a lock that AppControllers can use to have exclusive write access
 
187   # (between other AppControllers) to the ZooKeeper hierarchy located at
 
188   # APPCONTROLLER_PATH. It returns a boolean that indicates whether or not

217   # This method provides callers with an easier way to read and write to
 
218   # AppController data in ZooKeeper. This is useful for methods that aren't
 
219   # sure if they already have the ZooKeeper lock or not, but definitely need
 
220   # it and don't want to accidentally cause a deadlock (grabbing the lock when
 
221   # they already have it).
 
222   def self.lock_and_run(&block)
 
223     # Create the ZK lock path if it doesn't exist.
 
224     if !self.exists?(APPCONTROLLER_PATH)
 
225       self.set(APPCONTROLLER_PATH, DUMMY_DATA, NOT_EPHEMERAL)
 
226     end
 
227 
228     # Try to get the lock, and if we can't get it, see if we already have
 
229     # it. If we do, move on (but don't release it later since this block
 
230     # didn't grab it), and if we don't have it, try again.
 
231     got_lock = false
 
232     begin
 
233       if self.get_appcontroller_lock()
 
234         got_lock = true
 
235       else  # it may be that we already have the lock
 
236         info = self.run_zookeeper_operation {
 
237           @@zk.get(:path => APPCONTROLLER_LOCK_PATH)
 
238         }
 
239         owner = JSON.load(info[:data])
 
240         if @@client_ip == owner
 
241           NeptuneManager.log("Tried to get the lock, but we already have it. " +
 
242             "Not grabbing/releasing lock again.")
 
243           got_lock = false
 
244         else 
245           NeptuneManager.log("Tried to get the lock, but it's currently owned " +
 
246             "by #{owner}. Will try again later.")
 
247           raise ZooKeeperLockException
 
248         end
 
249       end
 
250     rescue ZooKeeperLockException => e
 
251       trace = e.backtrace.join("\n")
 
252       NeptuneManager.log("Saw an exception of class #{e.class}, with " +
 
253         "trace info: #{trace}")
 
254       Kernel.sleep(5)
 
255       retry
 
256     end
 
257 
258     begin
 
259       yield  # invoke the user's block, and catch any uncaught exceptions
 
260     #rescue Exception => except
 
261     #  NeptuneManager.log("Ran caller's block but saw an Exception of class " +
 
262     #    "#{except.class}")
 
263     #  raise except
 
264     ensure
 
265       if got_lock
 
266         NeptuneManager.log("Grabbed lock, so now releasing it.")
 
267         self.release_appcontroller_lock()
 
268         NeptuneManager.log("Released lock successfully")
 
269       else
 
270         NeptuneManager.log("Didn't grab lock, so not releasing it.")
 
271       end
 
272     end
 
273   end

538   def self.add_roles_to_node(roles, node)
