617   def get_n_items_of_work(n, queues)
 
618     items = []
 
619 
620     queues.each { |q|
 
621       loop {
 
622         NeptuneManager.log("popping an item of work off of queue #{q}")
 
623         start_time = Time.now
 
624         new_item = q.pop
 
625         end_time = Time.now
 
626         if new_item.nil?
 
627           NeptuneManager.log("#{q} is empty - moving on to next queue")
 
628           break  # the queue is empty
 
629         end
 
630     
631         # add how long it took to grab the item from the queue to our metadata
 
632         if !new_item["@metadata_info"]
 
633           new_item = {}
 
634         end
 
635         queue_pop_time = end_time - start_time
 
636         new_item['@metadata_info']['queue_pop_time'] = queue_pop_time
 
637 
638         NeptuneManager.log("adding [#{new_item}][#{new_item.class}] to items" +
 
639           " - took #{queue_pop_time} seconds to pop it off the queue")
 
640         items << new_item
 
641 
642         break if items.length >= n
 
643       }
 
644 
645       break if items.length >= n
 
646     }
 
647 
648     NeptuneManager.log("returning [#{items.join(', ')}] items of work")
 
649     return items

653   # A Babel master can call this method to add more workers (Babel slaves) to the
 
654   # system as needed.
 
655   def spawn_babel_slaves(num_of_waiting_tasks)
 
656     NeptuneManager.log("spawning workers to handle #{num_of_waiting_tasks} tasks")
 
657 
658     if is_cloud?
 
659       instance_type = "m1.large"
 
660       #instance_type = "m2.4xlarge"
 
661       cores_per_machine = INSTANCE_CPU_INFO[instance_type]
 
662     else
 
663       instance_type = "m1.large"
 
664       cores_per_machine = 2
 
665     end
 
666     NeptuneManager.log("Using #{cores_per_machine} cores/machine")
 
667 
668     # First, calculate how many machines we would need to run all the tasks
 
669     # as fast as possible, by running one task per core.
 
670     optimal_num_of_vms = (num_of_waiting_tasks / Float(cores_per_machine)).ceil
 
671     NeptuneManager.log("Will run #{num_of_waiting_tasks} on " +
 
672       "#{optimal_num_of_vms} VMs")
 
673 
674     # The optimal number of VMs is optimal with respect to performance, but
 
675     # is not optimal with respect to cost. As the user has told us what the
 
676     # maximum number of machines they want to run are, grab that number and
 
677     # compare the two to see how many machines we should actually acquire.
 
678     user_num_of_vms = ZKInterface.get_max_machines_for_babel_slaves()
 
679 
680     total_num_of_vms_needed = [optimal_num_of_vms, user_num_of_vms].min
 
681     NeptuneManager.log("For #{num_of_waiting_tasks} tasks, the optimal number of " +
 
682       "VMs to use is #{optimal_num_of_vms} VMs, while the user specified that " +
 
683       "no more than #{user_num_of_vms} VMs - using #{total_num_of_vms_needed} " +
 
684       "total VMs")
 
685 
686     # A previous invocation of this function may have already spawned up
 
687     # babel slaves, so to obey the maximum that the user has given us, we have to
 
688     # subtract any already running babel slaves out of the value calculated above.
 
689     babel_slaves_already_running = 0
 
690     @nodes.each { |node|
 
691       babel_slaves_already_running += 1 if node.is_babel_slave?
 
692     }
 
693 
694     num_of_vms_needed = total_num_of_vms_needed - babel_slaves_already_running
 
695 
696     if num_of_vms_needed > 0
 
697       NeptuneManager.log("#{babel_slaves_already_running} VMs are already running " +
 
698         "as babel slaves, so we still need to spawn #{num_of_vms_needed} VMs")
 
699 
700       # Include rabbitmq_slave in here since we want to always be able to point
 
701       # our RabbitMQ client to localhost to get tasks
 
702       nodes_needed = []
 
703       num_of_vms_needed.times { |i|
 
704         nodes_needed << ["rabbitmq_slave", "babel_slave"]
 
705       }
 
706 
707       start_new_roles_on_nodes(nodes_needed, instance_type,
 
708         HelperFunctions.get_secret())
 
709     else
 
710       NeptuneManager.log("#{babel_slaves_already_running} VMs are already " +
 
711         "running, and as can only have a maximum of #{num_of_vms_needed}, we " +
 
712         "don't need to acquire more babel slaves right now")
 
713     end
 
714 
715     return
 
716   end
