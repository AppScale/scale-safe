896 =begin
 
897   Hill Climbing Algorithm
 
898     # find minimum execution time t1
 
899     # find neighbors t0 and t2
 
900 
901     # if t0 is too low set it to t1
 
902     # if t2 is too high set it to t1
 
903 
904     # if no data for either, choose t2
 
905     # if no data for t0, choose t0
 
906     # if data for both, choose t1
 
907 =end
 
908   def optimal_nodes_hill_climbing(job_data, thing_to_optimize)
 
909     job_name = get_job_name(job_data)
 
910 
911     if thing_to_optimize != "cost" and thing_to_optimize != "performance"
 
912       abort("bad thing to optimize - can be cost or performance but was #{thing_to_optimize}")
 
913     end
 
914 
915     current_data = @jobs[job_name]
 
916     if current_data.nil? or current_data.empty?
 
917       NeptuneManager.log("neptune - no job data yet for [#{job_name}]")
 
918       return job_data["@nodes_to_use"]
 
919     end
 
920 
921     NeptuneManager.log("found job data for [#{job_name}]")
 
922 
923     min_val = INFINITY
 
924     optimal_job = nil
 
925     current_data.each { |job|
 
926       NeptuneManager.log("current job data is [#{job}]")
 
927 
928       if thing_to_optimize == "performance"
 
929         my_val = job.total_time
 
930       elsif thing_to_optimize == "cost"
 
931         my_val = job.cost
 
932       else
 
933         abort("bad thing to optimize again")
 
934       end
 
935 
936       if my_val < min_val
 
937         NeptuneManager.log("found a new minimum - [#{job}]")
 
938         optimal_job = job
 
939       end
 
940     }
 
941 
942     NeptuneManager.log("minimum is - [#{optimal_job}]")
 
943 
944     search_space = job_data["@can_run_on"]
 
945     t1 = optimal_job.num_nodes
 
946 
947     NeptuneManager.log("optimal right now is t1 = #{t1}")
 
948     t0, t2 = find_neighbors(t1, search_space)
 
949     NeptuneManager.log("t1's neighbors are #{t0} and t2 = #{t2}")
 
950 
951     d0 = get_job_data(job_name, t0)
 
952     d2 = get_job_data(job_name, t2)
 
953 
954     return t2 if d0.nil? and d2.nil?
 
955     return t0 if d0.nil?
 
956     return t1
 
957   end

960   def find_neighbors(val, search_space)
 
961     abort("no empty arrays") if search_space.nil? or search_space.empty?
 
962 
963     left, right = nil, nil
 
964     length = search_space.length
 
965     search_space.each_with_index { |item, index|
 
966       # set left
 
967       if index <= 0
 
968         left = val
 
969       else
 
970         left = search_space[index-1]
 
971       end
 
972 
973       # set right
 
974       if index < length - 1
 
975         right = search_space[index+1]
 
976       else
 
977         right = val
 
978       end
 
979 
980       break if item == val
 
981     }
 
982 
983     return left, right
 
984   end
